% Section 3 — The category of representations
\section{The category of representations}
\label{sec:category}

% Foundations: categories, functors, natural transformations,
% sections/retractions/adjunctions; the category of representations with the
% C-ladder transforms as morphisms; parity cast as a retraction and the
% round-trip defect as failure of a retraction.
%
% C-ladder:
%   C1  = plain source, C2 = Raw AST, C3 = Typed AST, C4 = Contract IR,
%   C1+ = annotated source PROJECTED FROM C4 (the information-parity control).

This section asks whether the categorical language buys anything the
measure-theoretic and information-theoretic machinery does not already supply.
Its purpose is to make the question \emph{decidable}: we build just enough
category theory to state precisely what ``$\ladderplus$ is the correct
control'' and ``the round-trip $\roundtrip{\ladder{4}}{\ladderplus}$ measures a
defect'' would mean as categorical statements, and we fix a criterion
(\cref{subsec:cat-gonogo}) the framing must meet to earn its place. We follow
\citet{maclane1998}, \citet{riehl2016}, and \citet{awodey2010}.

\subsection{Categories, functors, natural transformations}
\label{subsec:cat-basics}

\begin{definition}[Category]
\label{def:category}
A \emph{category} $\mathcal{C}$ consists of objects, a set
$\mathcal{C}(A,B)$ of morphisms for each ordered pair, an associative
composition $\circ$, and an identity $\mathrm{id}_A$ at each object acting as
a unit for composition.
\end{definition}

\begin{definition}[Functor, natural transformation]
\label{def:functor}
A \emph{functor} $F:\mathcal{C}\to\mathcal{D}$ sends objects to objects and
morphisms to morphisms, preserving identities and composition. For functors
$F,G:\mathcal{C}\to\mathcal{D}$, a \emph{natural transformation}
$\eta:F\Rightarrow G$ is a family of morphisms
$\eta_A:F(A)\to G(A)$ such that
$G(f)\circ\eta_A=\eta_B\circ F(f)$ for every $f:A\to B$ (the naturality
square commutes).
\end{definition}

These are the bookkeeping objects; the load-bearing notion for parity is the
next one.

\subsection{Sections, retractions, adjunctions}
\label{subsec:retraction}

\begin{definition}[Section, retraction, retract]
\label{def:retract}
Let $e:B\to A$ and $p:A\to B$ be morphisms with
\begin{equation}
  \label{eq:retract}
  p\circ e \;=\; \mathrm{id}_B.
\end{equation}
Then $e$ is a \emph{section} (it has a left inverse), $p$ is a
\emph{retraction} (it has a right inverse), and $B$ is a \emph{retract} of
$A$, written $B \retract A$. The composite $e\circ p:A\to A$ is then
idempotent ($(e\circ p)\circ(e\circ p)=e\circ p$); it is the identity on $A$
iff $e,p$ are mutually inverse isomorphisms.
\end{definition}

The asymmetry in \cref{def:retract} is exactly the asymmetry between the two
grades of parity in \cref{sec:parity}. A retract is ``$B$ sits inside $A$ and
can be recovered from it'' --- one composite is the identity ($p\circ
e=\mathrm{id}_B$) while the other ($e\circ p$) need not be. An isomorphism
collapses both. We will read \emph{operational} parity (\cref{def:parity}) as
a retraction and \emph{absolute} parity as an isomorphism, so that
\cref{prop:abs-strict} --- absolute is strictly stronger --- becomes the
categorical truism that an isomorphism is a retraction but not conversely.

\begin{definition}[Adjunction]
\label{def:adjunction}
Functors $L:\mathcal{C}\to\mathcal{D}$ and $R:\mathcal{D}\to\mathcal{C}$ form
an \emph{adjunction} $L\dashv R$ if there is a natural bijection
$\mathcal{D}(L A, D)\cong\mathcal{C}(A, R D)$. We record adjunctions only
because the projection/re-parse pair below is a candidate adjoint pair; whether
that structure earns its keep is part of the criterion of
\cref{subsec:cat-gonogo}.
\end{definition}

\subsection{The category $\catRep$ of representations}
\label{subsec:catrep}

\begin{definition}[$\catRep$]
\label{def:catrep}
Let $\catRep$ have as objects the representations of \cref{def:rep} (each a
measurable map $\rep:\Xspace\to\Zspace_{\rep}$, identified with its induced
sub-$\sigma$-algebra $\repsig$ up to $\Prob$-null sets), and as morphisms
$\rep\to\rep'$ the measurable \emph{transforms} $t$ with
$\rep' = t\circ\rep$ almost surely --- i.e.\ the ways to compute $\rep'$ from
$\rep$ without consulting the raw input. Composition and identities are
inherited from measurable maps.
\end{definition}

A morphism $\rep\to\rep'$ exists precisely when $\rep'$ is a measurable
function of $\rep$, i.e.\ when $\sigalg{\rep'}\subseteq\repsig$ (Doob--Dynkin,
\cref{subsec:rep}). So $\catRep$ is a refinement order made into a category:
arrows point from finer to coarser representations, and the existence of an
arrow is the ``can be computed downstream'' relation that the
data-processing inequality (\cref{cor:dpi-rep}) is monotone along.

The C-ladder of \texttt{preprint-v1} (Constitution §14.1) populates $\catRep$
with concrete objects and morphisms:
\[
  \begin{gathered}
  \ladder{1}\ \text{(plain source)},\quad
  \ladder{2}\ \text{(Raw AST)},\quad
  \ladder{3}\ \text{(Typed AST)},\\[2pt]
  \ladder{4}\ \text{(Contract IR)},\quad
  \ladderplus\ \text{(annotated source, projected from }\ladder{4}\text{)},
  \end{gathered}
\]
with the ladder transforms (parse, type, lift-to-contract, project,
re-parse) as morphisms. The control $\ladderplus$ is, by construction, the
\emph{projection of $\ladder{4}$ back to source form}: this is the morphism
the next subsection names.

\subsection{Parity as a retraction; the round-trip defect}
\label{subsec:retraction-parity}

The control was designed so that $\ladderplus$ carries the same predictive
content as $\ladder{4}$ while being rendered in surface (source) form. Two
ladder morphisms formalize the construction:
\[
  \begin{aligned}
  p&:\ladder{4}\longrightarrow\ladderplus
  &&(\text{project the Contract IR to annotated source}),\\
  e&:\ladderplus\longrightarrow\ladder{4}'
  &&(\text{re-parse the annotated source to IR}).
  \end{aligned}
\]

\begin{definition}[Categorical parity and round-trip defect]
\label{def:cat-parity}
Say $\ladderplus$ is in \emph{categorical parity} with $\ladder{4}$ if it is a
retract of $\ladder{4}$ witnessed by $(e,p)$: $p\circ e=\mathrm{id}_{\ladderplus}$
(\cref{eq:retract}). The \emph{round-trip defect} is the failure of the other
composite to be the identity,
\[
  \roundtrip{\ladder{4}}{\ladderplus}:\quad
  e\circ p \;=\; \ladder{4}\xrightarrow{\,p\,}\ladderplus\xrightarrow{\,e\,}\ladder{4}',
  \qquad
  \text{defect} \;\defeq\; \big[\,e\circ p \ne \mathrm{id}_{\ladder{4}}\,\big].
\]
\end{definition}

This reproduces, in categorical terms, the operational-vs-absolute split of
\cref{sec:parity}. The retraction equation $p\circ e=\mathrm{id}_{\ladderplus}$
says the annotated-source control loses nothing that the round trip through it
can detect --- the formal content of ``$\ladderplus$ is the correct control.''
The defect $e\circ p\ne\mathrm{id}_{\ladder{4}}$ says $\ladder{4}$ carries
byte-level structure that the projection discards and the re-parse does not
restore. The idempotent $e\circ p$ projects $\ladder{4}$ onto the part of its
structure that survives the round trip; its image is the parity-relevant
content and its ``kernel'' is the defect. The round-trip recovery data
(\cref{sec:parity-defect}: byte-exact AST recovery $0/30$, but contracts
$100\%$, node kinds $99.8\%$, types $96.4\%$) are exactly the empirical
profile of this idempotent --- near-identity on semantic structure, far from
identity on surface bytes.

\begin{remark}[What is and is not claimed here]
\label{rem:cat-claim}
\cref{def:cat-parity} is a \emph{framing}, not yet a theorem. It commits to
reading parity as a retraction and the defect as the non-identity composite,
and it lines those up with \cref{def:parity} and the round-trip data. Whether this
buys a proposition beyond \cref{sec:parity} --- for instance, that the
idempotent $e\circ p$ splits $\ladder{4}$ into a parity-relevant retract plus a
predictively inert complement, giving a categorical decomposition the
information-theoretic account does not already provide --- is the question
settled affirmatively in \cref{subsec:cat-theorem} (\cref{thm:retraction}).
\end{remark}

\subsection{When the categorical framing earns its place}
\label{subsec:cat-gonogo}

To keep this decision honest rather than aesthetic, we fix the criterion now,
before attempting the proof.

\begin{quote}
\textbf{Criterion (depth of category theory).} Section~\ref{sec:category}
survives as a section iff a proof establishes a proposition that (i) is stated
in the retraction language of \cref{def:cat-parity}, and (ii) is \emph{not} a
notational re-encoding of \cref{thm:irrelevance} or
\cref{prop:suff-iff-mi} --- i.e.\ it must either constrain the round-trip
defect in a way \cref{sec:parity} does not, or supply the splitting of
\cref{rem:cat-claim} as a usable decomposition feeding \cref{sec:parity-defect}.
If only a restatement is available, category theory is demoted: \cref{def:catrep}
and \cref{def:cat-parity} collapse into a single remark inside
\cref{sec:parity}, and this section is deleted.
\end{quote}

\noindent The foundations above are deliberately minimal --- just the
retraction vocabulary and the one category $\catRep$ --- so that demotion, if
it comes, costs one remark rather than a rewrite. This is the section's
insurance against decorative math.

\subsection{What the framing earns: the defect is a composable idempotent}
\label{subsec:cat-theorem}

This subsection discharges the criterion of \cref{subsec:cat-gonogo}. We
isolate the one proposition the retraction language buys, show it is stated in
the vocabulary of \cref{def:cat-parity}, and show it is not a re-encoding of
\cref{thm:irrelevance} or \cref{prop:suff-iff-mi} --- on the contrary, the
object it controls is provably invisible to both. The full proofs are in
\cref{app:retraction}.

\begin{definition}[Parity rung]
\label{def:parity-rung}
A \emph{parity rung} is a pair of $\catRep$-morphisms
$(p:A\to B,\ e:B\to A)$ with $p\circ e=\mathrm{id}_B$ (\cref{eq:retract}), so
that $B\retract A$ is witnessed by $(e,p)$, together with the standing
hypothesis (\cref{tab:parity-ledger}) that the endpoints are in operational
parity, $A\parityeq B$ --- a rung is the categorical encoding of
\cref{def:parity}. Its \emph{round trip} is the endomorphism
$\phi\defeq e\circ p:A\to A$; the rung is in \emph{absolute} parity when
$\phi=\mathrm{id}_A$ and in \emph{strictly operational} parity when
$\phi\ne\mathrm{id}_A$ (\cref{def:cat-parity}). The control pair
$\big(p:\ladder{4}\to\ladderplus,\ e:\ladderplus\to\ladder{4}\big)$ --- the
re-parse $e$ read as landing in the Contract-IR object $\ladder{4}$, its
deviation from a two-sided inverse being exactly the round-trip defect of
\cref{def:cat-parity} --- is the rung the paper instantiates.
\end{definition}

\begin{theorem}[Compositional retraction structure; information-invisibility of the defect]
\label{thm:retraction}
Let $(p,e)$ be a parity rung with round trip $\phi=e\circ p$.
\begin{enumerate}
  \item[(i)] \textup{(Idempotent splitting.)} $\phi$ is idempotent,
    $\phi\circ\phi=\phi$, and splits through $B$: the pair $(e,p)$ exhibits $B$
    as the retract onto which $\phi$ projects, so $B$ is, up to isomorphism, the
    image of $\phi$ --- the parity-relevant part of $A$. This is the splitting
    promised in \cref{rem:cat-claim}.
  \item[(ii)] \textup{(Compositionality.)} If $(p_1:A\to B,\,e_1)$ and
    $(p_2:B\to C,\,e_2)$ are parity rungs, then so is
    $(p_2\circ p_1,\ e_1\circ e_2)$, with round trip
    $\phi_{AC}=e_1\circ\phi_{BC}\circ p_1$ where $\phi_{BC}=e_2\circ p_2$.
    Operational parity is therefore closed under composition along the ladder,
    and $\phi_{AC}=\mathrm{id}_A$ iff both rungs are absolute
    ($e_1\circ p_1=\mathrm{id}_A$ and $\phi_{BC}=\mathrm{id}_B$).
  \item[(iii)] \textup{(Information-invisibility.)} Operational parity gives
    $\MI{Y}{A}=\MI{Y}{B}$ (\cref{prop:suff-iff-mi}) and
    $\bayesrisk(A)=\bayesrisk(B)$ (\cref{thm:irrelevance}); yet the defect may
    be nontrivial, $\phi\ne\mathrm{id}_A$ (\cref{prop:abs-strict}). The
    round-trip defect therefore lies in the common kernel of the two functionals
    the criterion forbids re-encoding: neither the mutual information of
    \cref{prop:suff-iff-mi} nor the Bayes risk of \cref{thm:irrelevance}
    separates $A$ from $B$, while the idempotent $\phi$ does.
\end{enumerate}
\idealization{Parts (i)--(ii) are purely categorical and assume neither
idealization. Part (iii) invokes \cref{thm:irrelevance} for
$\bayesrisk(A)=\bayesrisk(B)$ and so inherits its two idealizations of
\cref{rem:slogan} --- Bayes-optimality and exact operational parity; the claim
is that the defect is invisible to the \emph{at-optimum} functionals, not that
it is empirically zero (the quantitative question of \cref{sec:parity-defect}).}
\end{theorem}

\begin{corollary}[The defect domain, and subadditivity along the ladder]
\label{cor:defect-subadditive}
The round trip $\phi$ of \cref{thm:retraction} fixes the parity-relevant retract
$B$ and moves only its complement; the perturbation $\overline{v}$ that the
quantitative defect integrates (\cref{def:parity-defect},
\eqref{eq:parity-defect}) is exactly that complement --- the structure of $A$
that $\phi$ fails to fix. The categorical splitting thus names the object
\cref{sec:parity-defect} metrizes. Moreover, along any composite of parity rungs
$A=R_0\to\cdots\to R_m=B$ the predictive-law term is subadditive,
\[
  \Wass{1}\!\big(\preddist{R_0},\preddist{R_m}\big)
  \;\le\;
  \sum_{i=1}^{m}\Wass{1}\!\big(\preddist{R_{i-1}},\preddist{R_i}\big),
\]
each intermediate law $\preddist{R_i}$ being well defined precisely because
\cref{thm:retraction}(ii) makes every partial composite a genuine parity rung.
The triangle inequality is elementary; the categorical content is that
compositionality \emph{licenses} the chaining, so the C-ladder's defect
decomposes rung by rung rather than only in aggregate.
\end{corollary}

\begin{remark}[Why this passes the criterion; scope of the result]
\label{rem:cat-go-scope}
\cref{thm:retraction} meets both prongs of \cref{subsec:cat-gonogo}. It is
\emph{(i)} stated in the retraction language of \cref{def:cat-parity} --- the
idempotent $\phi=e\circ p$ and its splitting --- and \emph{(ii)} not a
notational re-encoding of \cref{thm:irrelevance} or \cref{prop:suff-iff-mi},
because \cref{thm:retraction}(iii) proves the object it controls is in the
\emph{kernel} of both: any statement those two results can make about $A$ and
$B$ is identical, so a proposition that distinguishes them cannot be a
restatement of either. It also discharges the alternative prong --- the
splitting of \cref{rem:cat-claim} as a decomposition feeding
\cref{sec:parity-defect} (\cref{cor:defect-subadditive}). What earns the section
is thus \cref{thm:retraction}(i),(iii) with \cref{cor:defect-subadditive}. The
compositionality of \cref{thm:retraction}(ii) is, for \texttt{preprint-v1}, a
structural result about a single instantiated rung
($\ladder{4}\rightleftarrows\ladderplus$); it becomes quantitatively
load-bearing only once the control is refined into several parity-controlled
rungs, and we mark its present use as the single-rung splitting it licenses, not
as a multi-rung measurement.
\end{remark}

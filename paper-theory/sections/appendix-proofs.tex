\subsection{The irrelevance theorem and its rate--distortion corollary}
\label{app:irrelevance}

The proof of \cref{thm:irrelevance} isolates two facts about Bayes risk that
are used again in \cref{cor:ratedist-gap}: that it is a functional of the
posterior kernel of \cref{prop:disint}, and that sufficiency collapses that
kernel onto the full-information posterior. We state each as a lemma, then
assemble the theorem and the corollary.

Throughout, for a representation $\rep$ we write
$q_{\rep} \defeq \post{\rep}(\rep(X),\cdot)$ for the regular conditional law of
$Y$ given $\rep(X)$ (\cref{prop:disint}), a $\mathcal{P}(\Yspace)$-valued random
element; $q_{\mathrm{id}} = \post{\mathrm{id}}(X,\cdot)$ is the full-information
posterior. For a posterior $q\in\mathcal{P}(\Yspace)$ define the
\emph{posterior Bayes loss}
\begin{equation}
  \label{eq:bayes-loss-functional}
  G(q) \;\defeq\; \inf_{a\in\Aspace}\,\int_{\Yspace}\loss(y,a)\,q(\mathrm{d}y),
\end{equation}
the smallest expected loss achievable against the belief $q$.

\begin{lemma}[Bayes risk is a concave functional of the posterior]
\label{lem:bayes-functional}
Let $\loss$ admit a measurable Bayes act. Then for every representation $\rep$,
\begin{equation}
  \label{eq:risk-as-posterior}
  \bayesrisk(\rep) \;=\; \Expect\big[\,G(q_{\rep})\,\big].
\end{equation}
The map $G$ is concave on $\mathcal{P}(\Yspace)$, and consequently
$\rep\mapsto\bayesrisk(\rep)$ is monotone under refinement: if
$\sigalg{\rep'}\subseteq\sigalg{\rep}$ up to $\Prob$-null sets then
$\bayesrisk(\rep)\le\bayesrisk(\rep')$.
\end{lemma}

\begin{proof}
Let $\kernel{K}$ be any predictor on $\rep$ (\cref{def:predictor}). Since
$\kernel{K}(\rep(X),\cdot)$ is $\sigalg{\rep}$-measurable, the tower rule
\eqref{eq:tower} applied to $\mathcal{G}_1=\sigalg{\rep}\subseteq
\mathcal{G}_2=\Bor{\Xspace}$ gives, for the risk of \cref{def:risk},
\[
  \risk(\kernel{K};\rep)
  = \Expect\!\Big[\int_{\Aspace}\loss(Y,a)\,\kernel{K}(\rep(X),\mathrm{d}a)\Big]
  = \Expect\!\Big[\int_{\Aspace}\!\Big(\!\int_{\Yspace}\!\loss(y,a)\,q_{\rep}(\mathrm{d}y)\Big)
        \kernel{K}(\rep(X),\mathrm{d}a)\Big]
  \;\ge\; \Expect\big[\,G(q_{\rep})\,\big],
\]
where the middle equality replaces $\Expect[\loss(Y,a)\mid\sigalg{\rep}]$ by the
posterior integral (\cref{prop:disint}) and the inequality bounds the inner
$a$-integral below by its infimum \eqref{eq:bayes-loss-functional}. Taking the
infimum over $\kernel{K}$ gives $\bayesrisk(\rep)\ge\Expect[G(q_{\rep})]$. For
the reverse inequality, let $z\mapsto a^{\star}(z)$ be a measurable Bayes act,
$a^{\star}(z)\in\arg\min_{a}\int\loss(y,a)\,\post{\rep}(z,\mathrm{d}y)$ (its
existence is the standing hypothesis, supplied by \cref{prop:disint}). The
deterministic predictor $\kernel{K}^{\star}(z,\cdot)=\delta_{a^{\star}(z)}$
attains $\risk(\kernel{K}^{\star};\rep)=\Expect[G(q_{\rep})]$, proving
\eqref{eq:risk-as-posterior}.

$G$ is an infimum of the affine maps $q\mapsto\int\loss(\cdot,a)\,\mathrm{d}q$,
hence concave. For monotonicity, $\sigalg{\rep'}\subseteq\sigalg{\rep}$ makes
$q_{\rep'}=\Expect[q_{\rep}\mid\sigalg{\rep'}]$ (the tower rule, applied to each
$B\in\Bor{\Yspace}$), so by conditional Jensen for the concave $G$,
$\Expect[G(q_{\rep})\mid\sigalg{\rep'}]\le G(q_{\rep'})$; taking expectations
and using \eqref{eq:risk-as-posterior} twice gives
$\bayesrisk(\rep)\le\bayesrisk(\rep')$.
\end{proof}

\begin{lemma}[Sufficiency collapses the posterior]
\label{lem:suff-collapse}
A representation $\rep$ is sufficient for $Y$ (\cref{def:sufficiency}) if and
only if $q_{\rep}=q_{\mathrm{id}}$ $\Prob$-a.s., i.e.\
$\post{\rep}(\rep(X),\cdot)=\post{\mathrm{id}}(X,\cdot)$ almost surely.
\end{lemma}

\begin{proof}
Sufficiency is the conditional independence $Y\indep X\given\rep(X)$
(\cref{def:sufficiency}), equivalently: for every $B\in\Bor{\Yspace}$ the
full-information posterior $\Prob(Y\in B\given\Bor{\Xspace})$ is
$\sigalg{\rep}$-measurable ($\Prob$-a.s.). Since
$\sigalg{\rep}\subseteq\Bor{\Xspace}$, the tower rule \eqref{eq:tower} gives
$\Prob(Y\in B\given\sigalg{\rep})=\Expect[\Prob(Y\in B\given\Bor{\Xspace})
\given\sigalg{\rep}]$; when the inner variable is itself
$\sigalg{\rep}$-measurable this conditional expectation returns it, so
$\Prob(Y\in B\given\Bor{\Xspace})=\Prob(Y\in B\given\sigalg{\rep})$ a.s. As
$\Yspace$ is standard Borel (\cref{ass:sbs}) the exceptional null set may be
chosen uniformly over a countable generating field for $\Bor{\Yspace}$ and the
identity extended to all $B$ by a monotone-class argument, yielding equality of
the regular conditional laws $q_{\mathrm{id}}=q_{\rep}$ a.s.\
\citep[Thm.~6.3]{kallenberg2002foundations}. Conversely, if
$q_{\rep}=q_{\mathrm{id}}$ a.s.\ then $q_{\mathrm{id}}$ is
$\sigalg{\rep}$-measurable, which is sufficiency.
\end{proof}

\begin{proof}[Proof of \cref{thm:irrelevance}]
By \cref{prop:suff-iff-mi}, the parity hypothesis $\rep\parityeq\rep'$
(\cref{def:parity}, the equality \eqref{eq:parity}) is exactly that both $\rep$
and $\rep'$ are sufficient for $Y$. By \cref{lem:suff-collapse} applied to each,
\[
  q_{\rep} \;=\; q_{\mathrm{id}} \;=\; q_{\rep'} \qquad \Prob\text{-a.s.}
\]
Apply the functional identity \eqref{eq:risk-as-posterior} of
\cref{lem:bayes-functional} to $\rep$, to $\mathrm{id}$, and to $\rep'$: the
three integrands $G(q_{\rep})$, $G(q_{\mathrm{id}})$, $G(q_{\rep'})$ coincide
a.s., so their expectations agree, giving
$\bayesrisk(\rep)=\bayesrisk(\mathrm{id})=\bayesrisk(\rep')$, which is
\eqref{eq:irrelevance}.
\idealization{The identity holds for the Bayes predictor $\post{\rep}$ of
\cref{prop:disint} (idealization~(i), Bayes-optimality) and under exact equality
\eqref{eq:parity} of mutual informations (idealization~(ii), exact parity); see
\cref{rem:slogan}. \cref{sec:bounded} relaxes~(i) and \cref{sec:parity-defect}
quantifies the failure of~(ii). No step uses absolute parity
(\cref{def:parity}); $\sigalg{\rep}=\sigalg{\rep'}$ is never invoked, only the
a.s.\ equality of posteriors that operational parity already supplies.}
\end{proof}

\begin{proof}[Proof of \cref{cor:ratedist-gap}]
Monotonicity $\bayesrisk(\rep)\ge\bayesrisk(\mathrm{id})$ is the refinement
inequality of \cref{lem:bayes-functional} with
$\sigalg{\rep}\subseteq\Bor{\Xspace}=\sigalg{\mathrm{id}}$; and
$\infogap{\rep}=0$ iff $\rep$ is sufficient by \cref{prop:suff-iff-mi}.

\emph{(a) Logarithmic loss.} With $\Aspace=\mathcal{P}(\Yspace)$ and
$\loss(y,q)=-\log q(y)$, for a fixed belief $p\in\mathcal{P}(\Yspace)$ the
posterior Bayes loss \eqref{eq:bayes-loss-functional} is
$G(p)=\inf_{q}\int(-\log q)\,\mathrm{d}p
=\inf_{q}\big(\Ent{p}+\KL{p}{q}\big)=\Ent{p}$, the infimum attained at $q=p$ by
Gibbs' inequality (\cref{def:kl}). Hence by \eqref{eq:risk-as-posterior}
$\bayesrisk(\rep)=\Expect[\Ent{q_{\rep}}]=\CondEnt{Y}{\rep(X)}$ and likewise
$\bayesrisk(\mathrm{id})=\CondEnt{Y}{X}$. Subtracting and using
$\CondEnt{Y}{U}=\Ent{Y}-\MI{Y}{U}$ (\cref{def:mi}) for $U\in\{\rep(X),X\}$,
\[
  \bayesrisk(\rep)-\bayesrisk(\mathrm{id})
  = \CondEnt{Y}{\rep(X)}-\CondEnt{Y}{X}
  = \MI{Y}{X}-\MI{Y}{\rep(X)}
  = \infogap{\rep},
\]
which is \eqref{eq:logloss-exact}.

\emph{(b) Bounded loss.} Write $q_{\rep}=\Expect[q_{\mathrm{id}}\mid
\sigalg{\rep}]$ (tower rule, as in \cref{lem:bayes-functional}). By
\eqref{eq:risk-as-posterior},
\[
  \bayesrisk(\rep)-\bayesrisk(\mathrm{id})
  = \Expect\big[G(q_{\rep})-G(q_{\mathrm{id}})\big]
  \;\le\; \Expect\big[\,\big|G(q_{\rep})-G(q_{\mathrm{id}})\big|\,\big].
\]
For any $q,q'\in\mathcal{P}(\Yspace)$, let $a'$ be a Bayes act for $q'$; then
$G(q)-G(q')\le\int\loss(\cdot,a')\,\mathrm{d}(q-q')\le\|\loss\|_\infty
\,\|q-q'\|_{\mathrm{TV}}$, where $\|q-q'\|_{\mathrm{TV}}=\sup_{B}|q(B)-q'(B)|$,
because $\loss(\cdot,a')\ge0$ has oscillation at most $\|\loss\|_\infty$; by
symmetry $|G(q)-G(q')|\le\|\loss\|_\infty\|q-q'\|_{\mathrm{TV}}$. Therefore
\[
  \bayesrisk(\rep)-\bayesrisk(\mathrm{id})
  \;\le\; \|\loss\|_\infty\,\Expect\big[\,\|q_{\rep}-q_{\mathrm{id}}\|_{\mathrm{TV}}\,\big].
\]
Pinsker's inequality
$\|q_{\mathrm{id}}-q_{\rep}\|_{\mathrm{TV}}\le\sqrt{\tfrac12\KL{q_{\mathrm{id}}}{q_{\rep}}}$
and Jensen's inequality for the concave square root give
$\Expect[\|q_{\mathrm{id}}-q_{\rep}\|_{\mathrm{TV}}]\le
\sqrt{\tfrac12\,\Expect[\KL{q_{\mathrm{id}}}{q_{\rep}}]}$. Finally, since
$\rep(X)$ is a measurable function of $X$, the chain rule \eqref{eq:chainrule}
gives $\MI{Y}{X}=\MI{Y}{\rep(X)}+\MI{Y}{X\given\rep(X)}$, whence
\begin{align*}
  \Expect\big[\KL{q_{\mathrm{id}}}{q_{\rep}}\big]
  &= \Expect\big[\KL{\post{\mathrm{id}}(X,\cdot)}{\post{\rep}(\rep(X),\cdot)}\big]
   = \MI{Y}{X\given\rep(X)} \\
  &= \MI{Y}{X}-\MI{Y}{\rep(X)}
   = \infogap{\rep},
\end{align*}
the second equality being the disintegrated form of conditional mutual
information \citep[Thm.~2.5.2 and eq.~(2.61)]{coverthomas2006}. Combining the
last three displays yields
$\bayesrisk(\rep)-\bayesrisk(\mathrm{id})\le\|\loss\|_\infty\sqrt{\tfrac12\infogap{\rep}}$,
which is \eqref{eq:ratedist-bound}; the lower bound $0$ is the monotonicity
already shown.
\idealization{Both bounds are statements about the \emph{Bayes} risk
(idealization~(i) retained); they relax exact parity (idealization~(ii)) to the
lossy case and make the failure quantitative through $\infogap{\rep}$. The
conversion of $\infogap{\rep}$ into a \emph{bounded}-predictor risk gap is not
established here; it is the go/no-go obligation of \cref{sec:parity-defect}
(T2-5). Identity~(a) is exact but specific to the log loss; bound~(b) is general
but, through Pinsker, loose by a square root --- neither is claimed to be the
risk gap of a real frozen LLM.}
\end{proof}

\subsection{The retraction proposition (T2-3)}
\label{app:retraction}

The category-theoretic claims are short once the retraction equation is in
hand; the only work is keeping the three parts honest about which hypothesis
each uses. Parts (i)--(ii) of \cref{thm:retraction} use only the equation
$p\circ e=\mathrm{id}_B$; part (iii) uses the operational-parity hypothesis of
\cref{def:parity-rung} and the already-proved \cref{thm:irrelevance,prop:suff-iff-mi,prop:abs-strict}.

\begin{proof}[Proof of \cref{thm:retraction}]
\emph{(i) Idempotent splitting.} Using $p\circ e=\mathrm{id}_B$ once,
\[
  \phi\circ\phi
  = (e\circ p)\circ(e\circ p)
  = e\circ(p\circ e)\circ p
  = e\circ\mathrm{id}_B\circ p
  = e\circ p
  = \phi,
\]
so $\phi$ is idempotent. An idempotent \emph{splits} through an object $I$ when
there are morphisms $s:I\to A$, $r:A\to I$ with $r\circ s=\mathrm{id}_I$ and
$s\circ r=\phi$; here $I=B$, $s=e$, $r=p$ satisfy $p\circ e=\mathrm{id}_B$ and
$e\circ p=\phi$. Hence $\phi$ splits through $B$, and in any category a split
idempotent identifies its splitting object with the image of the idempotent up
to canonical isomorphism, so $B\cong\operatorname{im}\phi$: $B$ is the retract
of $A$ that the round trip projects onto.

\emph{(ii) Compositionality.} For the composite,
\[
  (p_2\circ p_1)\circ(e_1\circ e_2)
  = p_2\circ(p_1\circ e_1)\circ e_2
  = p_2\circ\mathrm{id}_B\circ e_2
  = p_2\circ e_2
  = \mathrm{id}_C,
\]
so $(p_2\circ p_1,\ e_1\circ e_2)$ satisfies the retraction equation, and its
endpoints inherit operational parity by transitivity of \cref{def:parity}
($A\parityeq B\parityeq C$); it is a parity rung. Its round trip is
\[
  \phi_{AC}
  = (e_1\circ e_2)\circ(p_2\circ p_1)
  = e_1\circ(e_2\circ p_2)\circ p_1
  = e_1\circ\phi_{BC}\circ p_1 .
\]
For the equivalence: if $e_1\circ p_1=\mathrm{id}_A$ and
$\phi_{BC}=e_2\circ p_2=\mathrm{id}_B$ then
$\phi_{AC}=e_1\circ\mathrm{id}_B\circ p_1=e_1\circ p_1=\mathrm{id}_A$.
Conversely, suppose $\phi_{AC}=\mathrm{id}_A$. Pre- and post-composing with the
rung-1 maps and using $p_1\circ e_1=\mathrm{id}_B$ twice,
\[
  \phi_{BC}
  = (p_1\circ e_1)\,\phi_{BC}\,(p_1\circ e_1)
  = p_1\circ\big(e_1\circ\phi_{BC}\circ p_1\big)\circ e_1
  = p_1\circ\phi_{AC}\circ e_1
  = p_1\circ e_1
  = \mathrm{id}_B,
\]
and then $\mathrm{id}_A=\phi_{AC}=e_1\circ\mathrm{id}_B\circ p_1=e_1\circ p_1$,
so both rungs are absolute.

\emph{(iii) Information-invisibility.} By \cref{def:parity-rung} the endpoints
are in operational parity, $A\parityeq B$, which by \cref{prop:suff-iff-mi} is
$\MI{Y}{A(X)}=\MI{Y}{X}=\MI{Y}{B(X)}$ and by \cref{thm:irrelevance} gives
$\bayesrisk(A)=\bayesrisk(B)$. Both functionals depend on a representation only
through the predictive information it retains for $Y$ --- the mutual information
by definition, the Bayes risk through the posterior kernel of
\cref{lem:bayes-functional} --- and that quantity is equal for $A$ and $B$. The
defect, however, is the discrepancy of induced $\sigma$-algebras
$\sigalg{A}\ne\sigalg{B}$ (\cref{prop:abs-strict}), realized by the control rung
whose round trip fails byte-exact recovery (R4 data, $0/30$;
\cref{subsec:retraction-parity}); this is exactly the case $\phi\ne\mathrm{id}_A$.
A discrepancy of $\sigma$-algebras at equal retained information is not a
function of that information, so neither \cref{prop:suff-iff-mi} nor
\cref{thm:irrelevance} can register it, whereas $\phi$ does by construction. The
defect lies in their common kernel.
\idealization{Part (iii) inherits the two idealizations of \cref{rem:slogan}
through its use of \cref{thm:irrelevance}: Bayes-optimality and exact
operational parity. Parts (i)--(ii) are purely categorical and assume neither.
The proposition asserts the defect is invisible to the \emph{at-optimum}
functionals; it does not assert the defect is empirically zero --- that is the
quantitative question handed to \cref{sec:parity-defect} (T2-5).}
\end{proof}

\begin{proof}[Proof of \cref{cor:defect-subadditive}]
By \cref{thm:retraction}(i) the round trip $\phi$ acts as the identity on the
retract $B\cong\operatorname{im}\phi$ and nontrivially only off it; the
round-trip perturbation of \cref{def:cat-parity}, which
\eqref{eq:parity-defect} aggregates as $\overline{v}$, is therefore supported on
the complement of $\operatorname{im}\phi$ --- the structure $\phi$ fails to fix.
This identifies the domain of the \cref{sec:parity-defect} metric with the
categorical splitting. For subadditivity, \cref{thm:retraction}(ii) makes each
partial composite $R_0\to R_i$ a parity rung, so each $R_i$ is a $Y$-faithful
representation with a well-defined induced predictive law $\preddist{R_i}$
(\cref{subsec:defect-def}); the $1$-Wasserstein distance is a metric
(\cref{def:wasserstein}), so the triangle inequality applied along
$\preddist{R_0},\dots,\preddist{R_m}$ gives
$\Wass{1}(\preddist{R_0},\preddist{R_m})\le\sum_{i}\Wass{1}(\preddist{R_{i-1}},\preddist{R_i})$.
The bound is the elementary triangle inequality; the categorical content is that
(ii) supplies the intermediate laws that make the chain meaningful.
\end{proof}

\subsection{The bounded-learner advantage bound (T2-4)}
\label{app:bounded-gap}

The proof is a two-sided sandwich on the excess risk of the two representations,
using only the section's uniform-convergence bound (\cref{thm:uniform}) and the
irrelevance theorem (\cref{thm:irrelevance}). The single point of substance is
that the information channel cancels exactly, so the surviving bound is a pure
capacity quantity.

\begin{proof}[Proof of \cref{thm:bounded-gap}]
Parity $\rep\parityeq\rep'$ gives
$\bayesrisk(\rep)=\bayesrisk(\rep')=\bayesrisk(\mathrm{id})$
(\cref{thm:irrelevance}), so by \eqref{eq:approxerr} the in-class optima are
\begin{equation}
  \label{eq:inclass-optima}
  \inf_{f\in\hypclass\circ\rep}\risk(f)=\bayesrisk(\mathrm{id})+\approxerr{\rep},
  \qquad
  \inf_{f\in\hypclass\circ\rep'}\risk(f)=\bayesrisk(\mathrm{id})+\approxerr{\rep'}.
\end{equation}

\emph{Floor (deterministic).} The ERM $\widehat f_{\rep}$ lies in
$\hypclass\circ\rep$, so its risk is at least the class infimum; by
\eqref{eq:inclass-optima},
\begin{equation}
  \label{eq:floor}
  \risk(\widehat f_{\rep})\;\ge\;\inf_{f\in\hypclass\circ\rep}\risk(f)
  \;=\;\bayesrisk(\mathrm{id})+\approxerr{\rep},
\end{equation}
with no appeal to the sample.

\emph{Ceiling (high probability).} Apply \cref{thm:uniform} to
$\mathcal F=\hypclass\circ\rep'$: on an event of probability at least $1-\delta$
the ERM estimation error is at most
$4\,\Rad{\loss\circ(\hypclass\circ\rep')}+c_\nsample(\delta)$ with
$c_\nsample(\delta)=O(\sqrt{\log(1/\delta)/\nsample})$, so by
\eqref{eq:inclass-optima},
\begin{equation}
  \label{eq:ceiling}
  \risk(\widehat f_{\rep'})\;\le\;\inf_{f\in\hypclass\circ\rep'}\risk(f)
   +4\,\Rad{\loss\circ(\hypclass\circ\rep')}+c_\nsample(\delta)
  \;=\;\bayesrisk(\mathrm{id})+\approxerr{\rep'}
   +4\,\Rad{\loss\circ(\hypclass\circ\rep')}+c_\nsample(\delta).
\end{equation}

\emph{Subtraction.} On that event, by \cref{def:advantage} and
\eqref{eq:floor}--\eqref{eq:ceiling},
\[
  \advantage(\rep,\rep')
  =\risk(\widehat f_{\rep'})-\risk(\widehat f_{\rep})
  \;\le\;\big[\approxerr{\rep'}+4\,\Rad{\loss\circ(\hypclass\circ\rep')}
       +c_\nsample(\delta)\big]-\approxerr{\rep},
\]
the two $\bayesrisk(\mathrm{id})$ terms cancelling. This is
\eqref{eq:bounded-gap}.

\emph{No information term.} In the excess-risk decomposition \eqref{eq:excess}
the only information-dependent quantity is the middle gap
$\bayesrisk(\cdot)-\bayesrisk(\mathrm{id})=\infogap{\cdot}$; under parity it is
$0$ for each of $\rep,\rep'$ by \cref{thm:irrelevance}, so it never enters
\eqref{eq:floor}--\eqref{eq:ceiling}, and every term of \eqref{eq:bounded-gap}
is a capacity functional of an induced class, as claimed.
\idealization{The bound relaxes idealization~(i) of \cref{rem:slogan}
(Bayes-optimality) --- it concerns finite-sample ERMs, not the posterior
$\post{\rep}$ --- and retains idealization~(ii) (exact parity
\eqref{eq:parity}), used only to cancel $\infogap{\rep},\infogap{\rep'}$ via
\cref{thm:irrelevance}. It bounds the \emph{size} of any parity-preserving
advantage and is silent on sign and existence (guardrail~G2); the failure of
exact parity is quantified in \cref{sec:parity-defect}.}
\end{proof}

\subsection{The parity-defect risk bounds (T2-5)}
\label{app:parity-defect}

Both bounds are about the \emph{shared reader} $\kernel{K}$ of
\cref{subsec:defect-theorem}: a single predictor kernel evaluated at
$\ladder{4}(X)$ and at $\ladderplus(X)$. The global bound is Kantorovich--Rubinstein
duality applied per input and integrated; the local bound is the Pinsker/Fisher
chain of \cref{cor:ratedist-gap}(b) applied to the predictor's output family
rather than to the posterior. Neither uses Bayes-optimality, and the
information channel never appears --- it has already cancelled under parity.

\begin{proof}[Proof of \cref{thm:defect-risk-gap}]
Fix $x\in\Xspace$ and abbreviate the two output laws
$p_x\defeq\kernel{K}(\ladder{4}(x),\cdot)$ and
$p'_x\defeq\kernel{K}(\ladderplus(x),\cdot)$ on $(\Aspace,d)$. For every $y$ the
map $a\mapsto\loss(y,a)/L$ is $1$-Lipschitz, so applying the
Kantorovich--Rubinstein duality \eqref{eq:kr} to it and to its negation,
\begin{equation}
  \label{eq:per-instance-kr}
  \Big|\int_{\Aspace}\loss(y,a)\,p_x(\mathrm{d}a)
       -\int_{\Aspace}\loss(y,a)\,p'_x(\mathrm{d}a)\Big|
  \;\le\; L\,\Wass{1}(p_x,p'_x),
\end{equation}
a bound \emph{uniform in $y$}. Now expand the gap \eqref{eq:bp-gap} through the
risk functional \eqref{eq:risk-as-posterior}'s defining expectation
(\cref{def:risk}) and move the absolute value inside:
\[
  \big|\advantage_{\kernel{K}}(\ladder{4},\ladderplus)\big|
  = \Big|\Expect_{(X,Y)}\!\Big[\!\int\!\loss(Y,a)\,p_X(\mathrm{d}a)
        -\!\int\!\loss(Y,a)\,p'_X(\mathrm{d}a)\Big]\Big|
  \;\le\; \Expect_{(X,Y)}\Big[\,L\,\Wass{1}(p_X,p'_X)\,\Big],
\]
the inequality being Jensen for $|\cdot|$ followed by \eqref{eq:per-instance-kr}.
Since the integrand $L\,\Wass{1}(p_X,p'_X)$ does not depend on $Y$, the
expectation collapses to $\Expect_X$, giving
$\big|\advantage_{\kernel{K}}(\ladder{4},\ladderplus)\big|\le
L\,\Expect_X[\Wass{1}(p_X,p'_X)]=L\,\cpredtransport$ by
\eqref{eq:coupled-transport}, which is \eqref{eq:defect-risk-gap}.

For the cancellation claim, read the excess-risk decomposition \eqref{eq:excess}
for the \emph{same} kernel $\kernel{K}$ under each condition. The only
information-dependent summand is the middle gap
$\bayesrisk(\rep_c)-\bayesrisk(\mathrm{id})=\infogap{\rep_c}$; under
$\ladder{4}\parityeq\ladderplus$ both are $0$ (\cref{prop:suff-iff-mi}) and so
cancel in the difference \eqref{eq:bp-gap}, leaving only the
approximation/estimation discrepancy that \eqref{eq:defect-risk-gap} bounds by the
transport term. No mutual information survives.
\idealization{The bound concerns an arbitrary fixed bounded $\kernel{K}$, not the
Bayes posterior $\post{\rep}$ (idealization~(i) of \cref{rem:slogan} relaxed). It
uses operational parity only to cancel $\infogap{\ladder{4}}=\infogap{\ladderplus}=0$
(\cref{prop:suff-iff-mi}); it never invokes absolute parity, and is in fact
vacuous ($\cpredtransport=0$) under it. The conclusion is a magnitude bound,
sign- and existence-agnostic (guardrail~G2).}
\end{proof}

\begin{proof}[Proof of \cref{prop:defect-fisher-local}]
Fix $x$ and write $p_x,p'_x$ as above; let $a'$ be a minimizer-free generic
action. Exactly as in \cref{cor:ratedist-gap}(b), for a bounded loss the
expected-loss gap is controlled by total variation: choosing any reference act,
$|\!\int\loss(y,\cdot)\,\mathrm{d}(p_x-p'_x)|\le\|\loss\|_\infty\|p_x-p'_x\|_{\mathrm{TV}}$
uniformly in $y$, because $\loss(y,\cdot)$ has oscillation at most
$\|\loss\|_\infty$. Pinsker's inequality gives
$\|p_x-p'_x\|_{\mathrm{TV}}\le\sqrt{\tfrac12\KL{p_x}{p'_x}}$. Now parametrize the
output family along the round-trip perturbation. Both \cref{def:parity-defect} and
the estimator of \cref{subsec:defect-r4} operate at the \emph{aggregate} level: a
single mean increment $\overline{v}$ summarizes the round trip's structural
failures, and the Fisher term evaluates the output family's quadratic form at it.
Modeling the per-instance displacement by this aggregate, $p'_x$ is $p_x$
displaced by $\overline{v}$, so \cref{prop:kl-fisher} applied to the predictor's
output family at the operating point gives, for every $x$,
\[
  \KL{p_x}{p'_x}\;=\;\tfrac12\,\overline{v}^\top\Fisher\,\overline{v}\;+\;o(\|\overline{v}\|^2),
  \qquad\text{hence}\qquad
  \|p_x-p'_x\|_{\mathrm{TV}}\;\le\;\tfrac12\sqrt{\overline{v}^\top\Fisher\,\overline{v}}\;+\;o(\|\overline{v}\|).
\]
The right-hand side no longer depends on $x$ or $y$, so taking $\Expect_{(X,Y)}$
of the per-instance bound leaves it unchanged:
$\big|\advantage_{\kernel{K}}(\ladder{4},\ladderplus)\big|\le
\|\loss\|_\infty\,\tfrac12\sqrt{\overline{v}^\top\Fisher\,\overline{v}}+o(\|\overline{v}\|)$,
which is \eqref{eq:defect-fisher-local}.
\idealization{A bounded-predictor (idealization~(i) relaxed), \emph{local}
statement: the $o(\|\overline{v}\|)$ term is the second-order Taylor remainder of
\cref{prop:kl-fisher}, so the bound earns its keep only in the near-parity regime
$\|\overline{v}\|\to0$ where the global bound \eqref{eq:defect-risk-gap} may be
loose. It also models the per-instance round-trip increment by its aggregate
profile $\overline{v}$, the level at which \cref{def:parity-defect} and
\cref{subsec:defect-r4} are stated; input-level heterogeneity of the increment
would add a perturbation-covariance correction
$\tfrac14\|\loss\|_\infty^2\,\mathrm{tr}(\Fisher\,\mathrm{Cov}(v_X))$ of the same
(second) order, which the aggregate metric does not resolve. Sign-agnostic (G2).}
\end{proof}
